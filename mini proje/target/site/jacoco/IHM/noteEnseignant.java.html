public void matierComboBox() {
    Connection conn = connexion.getConnection();
        try {
        
        String query = "SELECT id, nom FROM matiere";
        PreparedStatement stmt = conn.prepareStatement(query);
        ResultSet rs = stmt.executeQuery();

        matierbox.removeAllItems(); // Vider avant de remplir

        while (rs.next()) {
            int id = rs.getInt("id");
            String nom = rs.getString("nom");
            matierbox.addItem(nom);
        }

        rs.close();
        stmt.close();
        conn.close();
    } catch (SQLException e) {
        e.printStackTrace();
        JOptionPane.showMessageDialog(null, "Erreur lors du chargement des branches");
    }
}
private void savenote() {try {
        Connection conn = connexion.getConnection();

    String valeur = jTextField1.getText().trim();
    String code = jTextField2.getText().trim();
    String matier = (String) matierbox.getSelectedItem();

                // Validate input fields
            if (valeur.isEmpty() || code.isEmpty()) {
                JOptionPane.showMessageDialog(this, "Veuillez remplir tous les champs", "Error", JOptionPane.ERROR_MESSAGE);
                return;
            }
            valeur = valeur.replaceAll("\\s+", "");
            code = code.replaceAll("\\s+", "");
            Note note = new Note();
            note.setCode(Integer.parseInt(code));
            note.setValeur(Integer.parseInt(valeur));

            
            int MatierId = getmatierIdFromName(matier, conn);
            note.setMatierid(MatierId);
            note dao = new note();
            boolean result = dao.savenote(note, conn);            
            if (result) {
                JOptionPane.showMessageDialog(this, "Note ajoute avec succès!", "Success", JOptionPane.INFORMATION_MESSAGE);
                clearnoteFields();
            } else {
                JOptionPane.showMessageDialog(this, "Erreur lors de la ajoute du note", "Error", JOptionPane.ERROR_MESSAGE);
            }
        } catch (Exception e) {
            JOptionPane.showMessageDialog(this, "Erreur: " + e.getMessage(), "Error", JOptionPane.ERROR_MESSAGE);
            e.printStackTrace();
        }}
    private void clearnoteFields() {
        jTextField1.setText("");
        jTextField2.setText("");
        matierbox.setSelectedIndex(0);
    }
    private int getmatierIdFromName(String matiereName, Connection conn) {
            PreparedStatement stmt = null;
            ResultSet rs = null;
            int id = 0;

            try {
                // Use the connection passed as parameter instead of creating a new one
                String query = "SELECT id FROM matiere WHERE nom = ?";
                stmt = conn.prepareStatement(query);
                stmt.setString(1, matiereName);
                rs = stmt.executeQuery();

                if (rs.next()) {
                    id = rs.getInt("id");
                }
            } catch (SQLException e) {
                e.printStackTrace();
                JOptionPane.showMessageDialog(null, "Erreur lors de la récupération de l'ID de la matier: " + e.getMessage());
            } finally {
                try {
                    if (rs != null) rs.close();
                    if (stmt != null) stmt.close();
                    // Do NOT close the connection here, as it will be used elsewhere
                } catch (SQLException e) {
                    e.printStackTrace();
                }
            }

            return id;
        }
