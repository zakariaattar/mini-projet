private void login() {
    String gmail = txtGmail.getText().trim();
    String password = txtPassword.getText().trim();

    if (gmail.isEmpty() || password.isEmpty()) {
        JOptionPane.showMessageDialog(this, "Veuillez remplir tous les champs", "Erreur", JOptionPane.ERROR_MESSAGE);
        return;
    }

    UserDAO userDAO = new UserDAO();

    if (userDAO.authenticateAdmin(gmail, password) != null) {
    JOptionPane.showMessageDialog(this, "Connexion réussie en tant qu'administrateur!");
    openAdminPanel(userDAO.getUserId()); // Erreur ici
    return;
    }

    if (userDAO.authenticateEnseignant(gmail, password) != null) {
    JOptionPane.showMessageDialog(this, "Connexion réussie en tant Enseignant!");
    openEnseignantPanel(userDAO.getUserId()); // Erreur ici
    return;        
    }

    if (userDAO.authenticateEtudiant(gmail, password) != null) {
    JOptionPane.showMessageDialog(this, "Connexion réussie en tant Etudiant!");
    openEtudiantPanel(userDAO.getUserId()); // Erreur ici
    return;         
    }

    // If none of the above conditions are met, the login has failed
    JOptionPane.showMessageDialog(this, "Gmail ou mot de passe invalide", "Erreur", JOptionPane.ERROR_MESSAGE);
}
